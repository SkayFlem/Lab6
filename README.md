является кратна ли его сумма цифр трем.

## 2. Реализация программы
#include <stdio.h>
#include <locale.h>

int main() {

    setlocale(LC_ALL,"RUS");
    int N, sum = 0;
    
    printf("Введите N (N < 1000): ");
    scanf("%d", &N);
    
    if (N >= 1000) {
        printf("Ошибка!\n");
        return 1;
    }

    sum = N/100 + (N/10)%10 + N%10;

    if (sum % 3 == 0)
        printf("Сумма цифр числа %d кратна 3\n", N);
    else
        printf("Сумма цифр числа %d не кратна 3\n", N);
    
    return 0;
}
